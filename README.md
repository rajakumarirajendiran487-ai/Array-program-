
# Shopping Cart using List ADT

cart = []

# Add items
cart.append("Rice")
cart.append("Milk")
cart.append("Bread")

print("Shopping Cart:", cart)

# Insert item
cart.insert(1, "Egg")

print("After Adding Egg:", cart)

# Remove item
cart.remove("Milk")

print("After Removing Milk:", cart)

# Display final cart
print("Final Cart Items:", cart)



Output:


Shopping Cart: ['Rice', 'Milk', 'Bread']
After Adding Egg: ['Rice', 'Egg', 'Milk', 'Bread']
After Removing Milk: ['Rice', 'Egg', 'Bread']
Final Cart Items: ['Rice', 'Egg', 'Bread']# Array-program-
